Generell

Diese Lightbox verwendet dieses Plugin. Es muss unterhalb von jQuery eingefügt werden.

Die Lightbox lässt sich so steuern: touch, pfeiltasten, auf den Rand des Bildes klicken bei Beispiel 3 auf Pfeile klicken. Schliessen kann man indem man einfach ausserhalb des Bildes klickt. Es lassen sich mehrere Lightboxen auf einer Seite einbinden. Sie müssen einfach einen eigenen selektor haben und einzeln initiiert werden.

Klick auf die Buttons öffnen die Beispiele.

// loader
var loaderOn = function () {
		$('body').append('
');
	},
	loaderOff = function () {
		$('.lightbox-loader').remove();
	},
 
// overlay
	overlayOn = function () {
		$('body').append('
');
	},
	overlayOff = function () {
		$('.lightbox-overlay').remove();
	},
 
// arrows
	arrowsOn = function (instance, selector) {
		var $arrows = $('leftright"');
		$arrows.appendTo('body');
		$arrows.on('click touchend', function (e) {
			e.preventDefault();
			var $this = $(this),
				$target = $(selector + '[href="' + $('#imagelightbox').attr('src') + '"]'),
				index = $target.index(selector);
 
			if ($this.hasClass('lightbox-left')) {
				index = index - 1;
				if (!$(selector).eq(index).length)
					index = $(selector).length;
			} else {
				index = index + 1;
				if (!$(selector).eq(index).length)
					index = 0;
			}
			instance.switchImageLightbox(index);
			return false;
		});
	},
	arrowsOff = function () {
		$('.lightbox-arrow').remove();
	};
 
// Initiating Lightboxes
$(document).ready(function(){
   // arrows, loader and overlay
	var selectorC = 'a[data-imagelightbox="c"]';
	var instanceC = $(selectorC).imageLightbox({
		onStart: function () { overlayOn(); arrowsOn(instanceC, selectorC);},
		onEnd: function () { overlayOff(); arrowsOff(); loaderOff();},
		onLoadStart: function () { loaderOn(); },
		onLoadEnd: function () { loaderOff(); $('.lightbox-arrow').css('display', 'block'); }
	});
	});
});
